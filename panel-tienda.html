// ══════════════════════════════════════ // NAVEGACIÓN // ══════════════════════════════════════ function showScreen(id) { document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.getElementById('screen-' + id).classList.add('active'); window.scrollTo(0,0); } function goTo(section) { if (section === 'dashboard') { updateDashboard(); showScreen('dashboard'); } else if (section === 'datos') { loadDatosForm(); showScreen('datos'); } else if (section === 'productos') { renderProducts(); showScreen('productos'); } else if (section === 'agregar') { resetAgregarForm(); showScreen('agregar'); } else if (section === 'password') { document.getElementById('pass-actual').value = ''; document.getElementById('pass-nueva').value = ''; document.getElementById('pass-repetir').value = ''; document.getElementById('pass-error').style.display = 'none'; document.getElementById('pass-ok').classList.remove('visible'); showScreen('password'); } } // ══════════════════════════════════════ // LOGIN / LOGOUT // ══════════════════════════════════════ function doLogin() { const email = document.getElementById('login-email').value.trim(); const pass = document.getElementById('login-pass').value; const err = document.getElementById('login-error'); if (email === DEMO_USER.email && pass === DEMO_USER.pass) { state.loggedIn = true; state.storeName = DEMO_USER.store; state.storeSlug = DEMO_USER.slug; err.classList.remove('visible'); updateDashboard(); showScreen('dashboard'); } else { err.classList.add('visible'); document.getElementById('login-pass').value = ''; } } function doLogout() { state.loggedIn = false; document.getElementById('login-email').value = ''; document.getElementById('login-pass').value = ''; showScreen('login'); } // ══════════════════════════════════════ // DASHBOARD // ══════════════════════════════════════ function updateDashboard() { document.getElementById('store-name-display').textContent = state.storeName; document.getElementById('store-link-display').textContent = 'minegocio.com.ar/' + state.storeSlug; const count = state.products.length; document.getElementById('products-count-desc').textContent = count === 0 ? 'Todavía no tenés productos cargados' : count === 1 ? '1 producto cargado' : count + ' productos cargados'; } function copyLink() { const link = 'https://minegocio.com.ar/' + state.storeSlug; navigator.clipboard.writeText(link).catch(() => {}); const btn = document.getElementById('copy-btn'); btn.textContent = '✓ Copiado'; btn.classList.add('copied'); setTimeout(() => { btn.textContent = 'Copiar'; btn.classList.remove('copied'); }, 2000); } // ══════════════════════════════════════ // DATOS DEL NEGOCIO // ══════════════════════════════════════ function loadDatosForm() { const d = state.datos; document.getElementById('d-nombre').value = d.nombre || ''; document.getElementById('d-rubro').value = d.rubro || ''; document.getElementById('d-descripcion').value = d.descripcion || ''; document.getElementById('d-whatsapp').value = d.whatsapp || ''; document.getElementById('d-zona').value = d.zona || ''; document.getElementById('d-pagos').value = d.pagos || ''; document.getElementById('datos-ok').classList.remove('visible'); } function saveDatos() { state.datos = { nombre: document.getElementById('d-nombre').value, rubro: document.getElementById('d-rubro').value, descripcion: document.getElementById('d-descripcion').value, whatsapp: document.getElementById('d-whatsapp').value, zona: document.getElementById('d-zona').value, pagos: document.getElementById('d-pagos').value, }; state.storeName = state.datos.nombre || state.storeName; document.getElementById('datos-ok').classList.add('visible'); setTimeout(() => document.getElementById('datos-ok').classList.remove('visible'), 2500); } // ══════════════════════════════════════ // PRODUCTOS // ══════════════════════════════════════ function renderProducts() { const list = document.getElementById('products-list'); const sub = document.getElementById('productos-sub'); if (state.products.length === 0) { list.innerHTML = `
📦

Todavía no cargaste productos.
¡Agregá el primero!

`; sub.textContent = ''; return; } sub.textContent = state.products.length + ' producto' + (state.products.length !== 1 ? 's' : '') + ' cargado' + (state.products.length !== 1 ? 's' : '') + '. Tocá Editar o Eliminar.'; list.innerHTML = state.products.map(p => `
${p.name}
$${Number(p.price).toLocaleString('es-AR')}
`).join(''); } function deleteProduct(id) { if (!confirm('¿Eliminar este producto?')) return; state.products = state.products.filter(p => p.id !== id); renderProducts(); updateDashboard(); } function editProduct(id) { const p = state.products.find(p => p.id === id); if (!p) return; state.editingId = id; state.pendingPhoto = p.photo; document.getElementById('agregar-title').textContent = 'Editar producto'; document.getElementById('p-nombre').value = p.name; document.getElementById('p-desc').value = p.desc || ''; document.getElementById('p-precio').value = p.price; const preview = document.getElementById('photo-preview'); if (p.photo) { preview.src = p.photo; preview.style.display = 'block'; } document.getElementById('product-ok').classList.remove('visible'); showScreen('agregar'); } function resetAgregarForm() { state.editingId = null; state.pendingPhoto = null; document.getElementById('agregar-title').textContent = 'Agregar producto'; document.getElementById('p-nombre').value = ''; document.getElementById('p-desc').value = ''; document.getElementById('p-precio').value = ''; document.getElementById('photo-preview').style.display = 'none'; document.getElementById('photo-preview').src = ''; document.getElementById('product-ok').classList.remove('visible'); } function saveProduct() { const name = document.getElementById('p-nombre').value.trim(); const desc = document.getElementById('p-desc').value.trim(); const price = document.getElementById('p-precio').value; if (!name || !price) { alert('Completá al menos el nombre y el precio.'); return; } if (state.editingId) { const idx = state.products.findIndex(p => p.id === state.editingId); if (idx >= 0) { state.products[idx] = { ...state.products[idx], name, desc, price: Number(price), photo: state.pendingPhoto || state.products[idx].photo }; } state.editingId = null; } else { state.products.push({ id: state.nextId++, name, desc, price: Number(price), photo: state.pendingPhoto || '', }); } state.pendingPhoto = null; document.getElementById('product-ok').classList.add('visible'); updateDashboard(); setTimeout(() => { document.getElementById('product-ok').classList.remove('visible'); resetAgregarForm(); goTo('dashboard'); }, 1200); } // ══════════════════════════════════════ // FOTO // ══════════════════════════════════════ function previewPhoto(input) { const file = input.files[0]; if (!file) return; const reader = new FileReader(); reader.onload = e => { const preview = document.getElementById('photo-preview'); preview.src = e.target.result; preview.style.display = 'block'; state.pendingPhoto = e.target.result; }; reader.readAsDataURL(file); } // ══════════════════════════════════════ // CAMBIAR CONTRASEÑA // ══════════════════════════════════════ function changePassword() { const actual = document.getElementById('pass-actual').value; const nueva = document.getElementById('pass-nueva').value; const repetir = document.getElementById('pass-repetir').value; const errEl = document.getElementById('pass-error'); errEl.style.display = 'none'; if (!actual || !nueva || !repetir) { errEl.textContent = 'Completá todos los campos.'; errEl.style.display = 'block'; return; } if (actual !== DEMO_USER.pass) { errEl.textContent = 'La contraseña actual es incorrecta.'; errEl.style.display = 'block'; return; } if (nueva.length < 6) { errEl.textContent = 'La nueva contraseña debe tener al menos 6 caracteres.'; errEl.style.display = 'block'; return; } if (nueva !== repetir) { errEl.textContent = 'Las contraseñas nuevas no coinciden.'; errEl.style.display = 'block'; return; } DEMO_USER.pass = nueva; document.getElementById('pass-actual').value = ''; document.getElementById('pass-nueva').value = ''; document.getElementById('pass-repetir').value = ''; document.getElementById('pass-ok').classList.add('visible'); setTimeout(() => { document.getElementById('pass-ok').classList.remove('visible'); goTo('dashboard'); }, 1800); }
